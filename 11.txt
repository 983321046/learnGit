git is a version control system.
git is free software
mkdir pp 创建目录 pp
cd pp 显示pp下的文件
pwd 展开路径
git init 创建git 仓库
git config --global user.name "用户名"
git congfig --global user.email "邮箱地址"
git add 文件
git commit -m "本次上传说明"
git status 监听文件夹状态
git diff 文件名  //查看区别

git reset --hard  head版本号开头几位  指定回到某版本
git reset --hard head^ 返回上个版本
cat 文件名  查看详细内容
git reflog 查看回档记录
git log 查看更新历史记录
